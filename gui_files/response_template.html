Your configuration file is not compatible with the GUI.

To do so, please add the following to your configuration file:

[USER-DEFINED]
CLOUDOPTION_NAME1 = cldattach model1 NAME1, vmcattach all
CLOUDOPTION_NAME2 = cldattach model2 NAME2, vmcattach all
...
CLOUDOPTION_NAMEX = cldattach modelX NAMEX, vmcattach all

[CATEGORY : CLOUDOPTION_NAMEX]
KEY = VALUE

# For each per-cloud category, suffix the category with the name of the cloud
# to which the request configuration options
# should belong.

# Where: # NAMEX = An arbitrary identifier chosen by you,
# one for as many clouds as you want to be able
# to use through the user interface.
# modelX = the model of the respective cloud (ec2, osk, scp, tac, etc...)

Example:

[USER-DEFINED]
CLOUDOPTION_OPENSTACK = cldattach osk OPENSTACK, vmcattach all
CLOUDOPTION_REDHAT = cldattach tac REDHAT, vmcattach all
CLOUDOPTION_OPENSTACK_OSK_INITIAL_VMCS = xxxxx
CLOUDOPTION_REDHAT_TAC_INITIAL_VMCS = xxxxx

# This gives your per-cloud-name flexibility
# within a single configuration file
[VM_DEFAULTS : CLOUDOPTION_OPENSTACK]
SOMEKEY = SOMEVALUE

[AI_TEMPLATES : CLOUDOPTION_REDHAT_DAYTRADER ]
SOMEKEY = SOMEVALUE
